# Comandos git

git init - inicializar o git

git clone - clonar repositorio

git remote add origin "nome do repositorio" - adiciona repositorio

git remote  remove origin - remove repositorio

git status - ver todas as alterações

git add . - adiciona todos os arquivos

git add "nome do arquivo"

git commit -m "Nome do commit" - "comita" as alterações

git push origin "nome da branch" - sobe as alterações pro repositorio remoto

git pull origin  "nome da branch" pegando as alterações 

git branch "nome da branch" - cria uma branch nova"

git checkout "nome da branch criada " vai pra branch nova

git checkout -b "nome da branch"

git config [user.name](http://user.name) <nome_usuario> - Configura um novo usuario

git config [user.email](http://user.email) <nome_usuario> - Configura um novo email
